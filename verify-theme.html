Theme Toggle Verification

Instructions:

  1. Open your Jekyll site at http://127.0.0.1:4000/forty-jekyll-theme/
  2. Open the browser developer console (F12)
  3. Copy and paste the test script below into the console
  4. Observe the test results in the console

Test Script:

// Quick verification script
(function() {
    console.log('=== Theme Toggle Verification ===\n');
    
    // Check button exists
    const button = document.getElementById('theme-toggle');
    console.log('✓ Theme toggle button:', button ? 'Found' : 'NOT FOUND');
    
    // Check if visible
    if (button) {
        const styles = getComputedStyle(button);
        const isVisible = styles.display !== 'none' && styles.visibility !== 'hidden';
        console.log('✓ Button visibility:', isVisible ? 'Visible' : 'NOT VISIBLE');
        console.log('  - Display:', styles.display);
        console.log('  - Visibility:', styles.visibility);
        console.log('  - Opacity:', styles.opacity);
    }
    
    // Check CSS variables
    const rootStyles = getComputedStyle(document.documentElement);
    console.log('\n✓ CSS Variables:');
    console.log('  - --bg:', rootStyles.getPropertyValue('--bg'));
    console.log('  - --fg:', rootStyles.getPropertyValue('--fg'));
    console.log('  - --highlight:', rootStyles.getPropertyValue('--highlight'));
    
    // Test theme toggle
    if (button) {
        console.log('\n✓ Testing theme toggle...');
        const initialTheme = document.documentElement.getAttribute('data-theme');
        console.log('  - Initial theme:', initialTheme || 'light');
        
        // Click button
        button.click();
        
        const newTheme = document.documentElement.getAttribute('data-theme');
        console.log('  - New theme:', newTheme || 'light');
        console.log('  - Toggle worked:', initialTheme !== newTheme ? 'YES' : 'NO');
        
        // Check localStorage
        console.log('  - Saved in localStorage:', localStorage.getItem('theme'));
    }
    
    console.log('\n=== End of Verification ===');
})();
